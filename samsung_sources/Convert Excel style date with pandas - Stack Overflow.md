You can use the 3rd party

http://xlrd.readthedocs.io/en/latest/

library before passing to

pd.to\_datetime

https://pandas.pydata.org/pandas-docs/stable/generated/pandas.to\_datetime.html

import xlrd

def read\_date(date):
    return xlrd.xldate.xldate\_as\_datetime(date, 0)

df = pd.DataFrame({'date':\[42580.3333333333, 10023\]})

df\['new'\] = pd.to\_datetime(df\['date'\].apply(read\_date), errors='coerce')

print(df)

           date                 new
0  42580.333333 2016-07-29 08:00:00
1  10023.000000 1927-06-10 00:00:00
